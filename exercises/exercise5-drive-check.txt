# get the drive check script (is in tarball, but not in packages)

cd
sudo wget https://github.com/rpedde/swift-training-kick/raw/master/scripts/swift-drive-audit

# copy the script to all the storage nodes
dsh -g storage sudo cp ~/swift-drive-audit /usr/bin

# make the config file
cat > ~/drive-audit.conf <<EOF
[drive-audit]
device_dir = /srv/node
log_facility = LOG_LOCAL0
log_level = INFO
minutes = 60
error_limit = 2
EOF

# copy the config to all the storage nodes
dsh -g storage cp ~/drive-audit.conf /etc/swift

# make a cron.d job
cat > ~/drive-audit <<EOF
15 * * * * root /usr/bin/swift-drive-audit
EOF

# copy the cron job to all the storage nodes
dsh -g storage sudo cp ~/drive-audit /etc/cron.d
