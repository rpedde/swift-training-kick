# show stats (containers, objects, etc) for test1 user
st -A http://localhost:8080/auth/v1.0/ -U testaccount:test1 -K test1pass stat

# create a container 
st -A http://localhost:8080/auth/v1.0/ -U testaccount:test1 -K test1pass post testcontainer

# verify the container exists
st -A http://localhost:8080/auth/v1.0/ -U testaccount:test1 -K test1pass list

# upload a file to the container
cat > test.txt <<EOF
This is a test file.
EOF

st -A http://localhost:8080/auth/v1.0/ -U testaccount:test1 -K test1pass upload testcontainer test.txt

# view the file
st -A http://localhost:8080/auth/v1.0/ -U testaccount:test1 -K test1pass download testcontainer test.txt -o -

# can we view the file as test2?
st -A http://localhost:8080/auth/v1.0/ -U testaccount:test2 -K test2pass download testcontainer test.txt -o -
